UUID SQL Formatting *dkuuid* *uuid* <<<

# PHP SQL String
"LCASE(CONCAT(" .
"  HEX(SUBSTR(`UUID`, 1, 4)), '-'," .
"  HEX(SUBSTR(`UUID`, 5, 2)), '-'," .
"  HEX(SUBSTR(`UUID`, 7, 2)), '-'," .
"  HEX(SUBSTR(`UUID`, 9, 2)), '-'," .
"  HEX(SUBSTR(`UUID`, 11))" .
")) `UUID`"

# SQL
LCASE(CONCAT(HEX(SUBSTR(`UUID`, 1, 4)), '-', HEX(SUBSTR(`UUID`, 5, 2)), '-', HEX(SUBSTR(`UUID`, 7, 2)), '-', HEX(SUBSTR(`UUID`, 9, 2)), '-', HEX(SUBSTR(`UUID`, 11)))) `UUID`

# Double quotes
LCASE(CONCAT(HEX(SUBSTR(`UUID`, 1, 4)), "-", HEX(SUBSTR(`UUID`, 5, 2)), "-", HEX(SUBSTR(`UUID`, 7, 2)), "-", HEX(SUBSTR(`UUID`, 9, 2)), "-", HEX(SUBSTR(`UUID`, 11)))) `UUID`
